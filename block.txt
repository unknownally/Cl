Code Program: Client Class with Hash Generation
import hashlib
import datetime

class Client:
    def __init__(self, name, data):
        self.name = name
        self.data = data
        self.timestamp = datetime.datetime.now()
        self.hash = self.generate_hash()

    def generate_hash(self):
        sha = hashlib.sha256()
        hash_string = str(self.name) + str(self.data) + str(self.timestamp)
        sha.update(hash_string.encode('utf-8'))
        return sha.hexdigest()

    def display(self):
        print("Client Name:", self.name)
        print("Data:", self.data)
        print("Timestamp:", self.timestamp)
        print("Generated Hash:", self.hash)

# Example
client1 = Client("Alma", "Transaction of 100 coins")
client1.display()

import hashlib
import time

class Block:
    def __init__(self, block_id, prev_hash, sender, receiver, amount):
        self.block_id = block_id
        self.timestamp = time.time()
        self.nonce = 0
        self.prev_hash = prev_hash

        # Single transaction
        self.transaction = {
            'sender': sender,
            'receiver': receiver,
            'amount': amount
        }

        self.merkle_root = self.calculate_merkle_root()
        self.hash = self.calculate_hash()

    def calculate_merkle_root(self):
        transaction_str = (
            str(self.transaction['sender']) +
            str(self.transaction['receiver']) +
            str(self.transaction['amount'])
        )
        return hashlib.sha256(transaction_str.encode()).hexdigest()

    def calculate_hash(self):
        block_str = (
            str(self.block_id) +
            str(self.timestamp) +
            str(self.nonce) +
            str(self.prev_hash) +
            str(self.merkle_root)
        )
        return hashlib.sha256(block_str.encode()).hexdigest()

    def display_block_details(self):
        print("\n---------------------------")
        print("Block ID:", self.block_id)
        print("Timestamp:", self.timestamp)
        print("Nonce:", self.nonce)
        print("Previous Hash:", self.prev_hash)
        print("Transaction:")
        print("\tSender:", self.transaction['sender'])
        print("\tReceiver:", self.transaction['receiver'])
        print("\tAmount:", self.transaction['amount'])
        print("Merkle Root:", self.merkle_root)
        print("Hash:", self.hash)

Title:To implement a simple block chain in Python with one transaction per block using hashing.


Code:
import hashlib
import time

# ------------------ Transaction Class ------------------
class Transaction:
    def __init__(self, sender, receiver, amount):
        self.sender = sender
        self.receiver = receiver
        self.amount = amount

    def __str__(self):
        return f"{self.sender} -> {self.receiver} : {self.amount}"

# ------------------ Block Class ------------------
class Block:
    def __init__(self, block_id, prev_hash, transactions):
        self.block_id = block_id
        self.timestamp = time.time()
        self.prev_hash = prev_hash
        self.transactions = transactions
        self.merkle_root = self.calculate_merkle_root()
        self.hash = self.calculate_hash()

    def calculate_merkle_root(self):
        tx_data = "".join(str(tx) for tx in self.transactions)
        return hashlib.sha256(tx_data.encode()).hexdigest()

    def calculate_hash(self):
        data = (
            str(self.block_id) +
            str(self.timestamp) +
            str(self.prev_hash) +
            str(self.merkle_root)
        )
        return hashlib.sha256(data.encode()).hexdigest()

    def display_block(self):
        print("\n-----------------------------")
        print(f"Block ID: {self.block_id}")
        print(f"Timestamp: {self.timestamp}")
        print(f"Previous Hash: {self.prev_hash}")
        print(f"Merkle Root: {self.merkle_root}")
        print(f"Hash: {self.hash}")
        print("Transactions:")
        for tx in self.transactions:
            print("  ", tx)

# ------------------ Blockchain Class ------------------
class Blockchain:
    def __init__(self):
        self.chain = []
        self.create_genesis_block()

    def create_genesis_block(self):
        genesis_tx = [Transaction("Genesis", "Alice", 100)]
        genesis_block = Block(0, "0", genesis_tx)
        self.chain.append(genesis_block)

    def add_block(self, transactions):
        prev_block = self.chain[-1]
        new_block = Block(len(self.chain), prev_block.hash, transactions)
        self.chain.append(new_block)

    def display_chain(self):
        for block in self.chain:
            block.display_block()

# ------------------ Main Program ------------------
def main():
    blockchain = Blockchain()

    num_blocks = int(input("Enter number of blocks to add: "))

    for i in range(num_blocks):
        print(f"\nEnter transactions for Block {i + 1}")
        num_tx = int(input("How many transactions? "))

        transactions = []
        for j in range(num_tx):
            sender = input("Sender: ")
            receiver = input("Receiver: ")
            amount = float(input("Amount: "))
            transactions.append(Transaction(sender, receiver, amount))

        blockchain.add_block(transactions)

    print("\n===== BLOCKCHAIN =====")
    blockchain.display_chain()

# Run program
if __name__ == "__main__":
    main()

Title:To write a Solidity program to demonstrate the use of arrays and their operations
// SPDX-License-Identifier: MIT
pragma solidity ^0.8.0;

contract ArrayOperations {

    uint[] public arr;

    // Add element to array
    function addElement(uint _value) public returns (uint[] memory) {
        arr.push(_value);
        return arr;
    }

    // Get full array
    function getArray() public view returns (uint[] memory) {
        return arr;
    }

    // Get element by index
    function getElement(uint _index) public view returns (uint) {
        require(_index < arr.length, "Index out of bounds");
        return arr[_index];
    }

    // Update element
    function updateElement(uint _index, uint _value) public returns (uint[] memory) {
        require(_index < arr.length, "Index out of bounds");
        arr[_index] = _value;
        return arr;
    }

    // Delete element (resets to 0)
    function deleteElement(uint _index) public returns (uint[] memory) {
        require(_index < arr.length, "Index out of bounds");
        delete arr[_index];
        return arr;
    }

    // Get array length
    function getLength() public view returns (uint) {
        return arr.length;
    }

    // Remove last element
    function popElement() public returns (uint[] memory) {
        require(arr.length > 0, "Array is empty");
        arr.pop();
        return arr;
    }
}
